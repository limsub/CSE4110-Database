create table shipper (shipper_ID varchar(5), name varchar(20), address varchar(20), account_num varchar(20), primary key (shipper_ID));
create table customer (customer_ID varchar(5), name varchar(20), address varchar(20), account_num varchar(20), primary key(customer_ID));
create table contract (shipper_ID varchar(5), customer_ID varchar(5), primary key (shipper_ID, customer_ID), foreign key (shipper_ID) references shipper (shipper_ID), foreign key (customer_ID) references customer (customer_ID) );
create table shipping_company (company_name varchar(20), contact_number varchar(20), primary key (company_name) );
create table user_shipper (company_name varchar(20), shipper_ID varchar(5), num_of_use numeric(5, 0), total_payment_cost numeric(12,2), primary key (company_name, shipper_ID), foreign key (company_name) references shipping_company (company_name), foreign key (shipper_ID) references shipper (shipper_ID) );
create table user_customer (company_name varchar(20), customer_ID varchar(5), num_of_use numeric(5, 0), total_payment_cost numeric(12,2), primary key (company_name, customer_ID), foreign key (company_name) references shipping_company (company_name), foreign key (customer_ID) references customer (customer_ID) );
create table bill (bill_ID varchar(5), payer_ID varchar(5), method varchar(20), price numeric(12,2), billing_date datetime, payment_date datetime, bill_status varchar(20), primary key (bill_ID) );
create table package (package_ID varchar(5), shipper_ID varchar(5), customer_ID varchar(5), company_name varchar(20), box_type varchar(10), weight numeric(5, 2), delivery_type varchar(20), deliver_status varchar(20), start_day datetime, end_day datetime, promised_day datetime, special_note varchar(20), bill_ID varchar(5), primary key (package_ID), foreign key (shipper_ID) references shipper (shipper_ID), foreign key (customer_ID) references customer (customer_ID), foreign key (company_name) references shipping_company (company_name), foreign key (bill_ID) references bill (bill_ID) );
create table driver (license_ID varchar(5), name varchar(20), phone_number varchar(20), primary key (license_ID) );
create table vehicle (vehicle_ID varchar(5), vehicle_type varchar(20), license_ID varchar(5), maximum_loads numeric(10,2), vehicle_status varchar(20), primary key (vehicle_ID, vehicle_type), foreign key (license_ID) references driver (license_ID) );
create table deliver (package_ID varchar(5), vehicle_ID varchar(5), vehicle_type varchar(20), primary key (package_ID), foreign key (package_ID) references package (package_ID), foreign key (vehicle_ID, vehicle_type) references vehicle (vehicle_ID, vehicle_type) );
create table warehouse (warehouse_address varchar(20), warehouse_name varchar(20), primary key (warehouse_address) );
create table start_point (vehicle_ID varchar(5), vehicle_type varchar(20), warehouse_address varchar(20), primary key (vehicle_ID, vehicle_type), foreign key (vehicle_ID, vehicle_type) references vehicle (vehicle_ID, vehicle_type), foreign key (warehouse_address) references warehouse (warehouse_address) );
create table destination (vehicle_ID varchar(5), vehicle_type varchar(20), warehouse_address varchar(20), primary key (vehicle_ID, vehicle_type), foreign key (vehicle_ID, vehicle_type) references vehicle (vehicle_ID, vehicle_type), foreign key (warehouse_address) references warehouse (warehouse_address) );
create table have_been (package_ID varchar(5), warehouse_address varchar(20), daytime datetime, primary key (package_ID, warehouse_address), foreign key (package_ID) references package (package_ID), foreign key (warehouse_address) references warehouse (warehouse_address) );
